2. core/slimmer.js
本脚本是注入到目标网页的 JavaScript，用于实现“语义感知”的 DOM 瘦身，移除冗余元素，确保提取的上下文不超限。
/**
 * DAILM 高性能 JavaScript 注入脚本 - 语义 DOM 瘦身
 * 目的：移除冗余元素，提取语义关键文本和交互元素。
 */
(() => {
    // 定义需要移除的不必要标签
    const JUNK_TAGS = ['script', 'style', 'link', 'header', 'footer', 'nav', 'iframe', 'noscript', 'img', 'svg', 'video', 'audio'];
    
    // 移除所有junk标签
    JUNK_TAGS.forEach(tag => {
        const elements = document.querySelectorAll(tag);
        elements.forEach(el => el.remove());
    });

    // 定义需要提取的核心文本和交互元素选择器
    const SEMANTIC_SELECTOR = 'p, h1, h2, h3, h4, h5, h6, ul, ol, li, blockquote, pre, code, button, input[type="text"], input[type="search"], textarea';
    
    // 提取所有语义元素并转换为精简结构
    const semanticElements = document.querySelectorAll(SEMANTIC_SELECTOR);
    let semanticDOM = [];

    semanticElements.forEach((el, index) => {
        let slimmedElement = {
            id: index,
            tag: el.tagName.toLowerCase(),
            text: el.innerText ? el.innerText.trim() : null
        };

        // 如果是交互元素，添加额外属性
        if (['button', 'input', 'textarea'].includes(slimmedElement.tag)) {
            slimmedElement.type = el.type || null;
            slimmedElement.name = el.name || null;
            slimmedElement.id_attr = el.id || null;
            slimmedElement.placeholder = el.placeholder || null;
        }

        // 仅保留有价值信息的元素
        if (slimmedElement.text || slimmedElement.placeholder) {
            semanticDOM.push(slimmedElement);
        }
    });

    // 将精简后的语义DOM序列化为JSON字符串
    return JSON.stringify(semanticDOM, null, 2);
})();

class BrowserEngine:
    def __init__(self, user_data_dir: str = "./user_data"):
        """
        初始化浏览器引擎。
        :param user_data_dir: 核心技术规范：持久化 Session 目录，用户手动登录后可绕过验证码。
        """
        self.user_data_dir = os.path.abspath(user_data_dir)
        self.playwright = None
        self.context = None
        logger.info(f"BrowserEngine initialized. UserDataDir: {self.user_data_dir}")

    async def start(self, headless: bool = False):
        """
        启动 Playwright 浏览器并加载持久化会话。
        :param headless: 是否使用无头模式。
        """
        self.playwright = await async_playwright().start()
        
        # 核心技术规范：启动持久化上下文防检测配置
        # args 中禁用了 AutomationControlled，隐藏 webdriver 特征
        self.context = await self.playwright.chromium.launch_persistent_context(
            user_data_dir=self.user_data_dir,
            headless=headless,
            # [逻辑阶梯约束：安全性] 随机延迟和真人路径模拟防检测
            args=[
                "--disable-blink-features=AutomationControlled", # 隐藏 Playwright 特征
                "--no-sandbox",
                "--disable-infobars",
                "--disable-dev-shm-usage"
            ],
            ignore_default_args=["--enable-automation"],
            viewport={'width': 1280, 'height': 720}
        )
        
        # 设置默认超时
        self.context.set_default_timeout(30000)
        logger.success("Persistent context launched successfully.")

    async def get_page(self) -> Page:
        """从上下文中获取新的页面。"""
        if self.context:
            page = await self.context.new_page()
            # 进一步隐藏webdriver特征
            await page.add_init_script("Object.defineProperty(navigator, 'webdriver', {get: () => undefined})")
            return page
        raise Exception("Browser context not started.")

    async def load_slimmer_script(self) -> str:
        """加载注入式 DOM 压缩脚本核心文件。"""
        script_path = os.path.join(os.path.dirname(__file__), "slimmer.js")
        with open(script_path, "r", encoding="utf-8") as f:
            return f.read()

    async def inject_slimmer(self, page: Page):
        """核心技术规范：多维度 DOM 瘦身脚本注入并执行。"""
        logger.info("Injecting Semantic DOM Slimmer script...")
        script = await self.load_slimmer_script()
        
        # 执行注入脚本并获取 JSON Handshake 数据
        semantic_json_str = await page.evaluate(script)
        logger.info("DOM Slimming complete. Context reduced successfully.")
        return semantic_json_str

    async def stop(self):
        """关闭 Playwright 浏览器及所有标签页。"""
        if self.context:
            await self.context.close()
        if self.playwright:
            await self.playwright.stop()
        logger.info("BrowserEngine stopped.")

本文件实现了 The Matrix Router，负责核心技术规范：总线调度协议和算力矩阵适配器的任务路由。所有通信均采用 JSON 格式的“握手协议”。

import uuid
import json
from datetime import datetime
from loguru import logger

class ModelMatrix:
    def __init__(self, adapters: dict):
        """
        核心技术规范：算力矩阵适配器调度类。
        它根据任务类型将 Prompt 路由到最合适的标签页。
        :param adapters: 一个字典，包含初始化后的适配器实例 (e.g., {'gemini': gemini_adapter_instance})。
        """
        self.adapters = adapters
        logger.info("The Matrix Router initialized with adapters.")

    def json_handshake(self, step_id, source_model, thought, action, data):
        """核心技术规范：JSON 握手协议通信格式生成器。"""
        return {
            "step_id": str(step_id),
            "timestamp": datetime.now().isoformat(),
            "source_model": source_model,
            "thought": thought,
            "action": action,
            "data": data
        }

    async def route_task(self, task_type: str, prompt: str) -> dict:
        """核心技术规范：总线调度协议实现。通过感知层将任务路由至执行层。"""
        step_id = uuid.uuid4()
        logger.info(f"Routing task (ID: {step_id}) - Type: {task_type}...")

        # 定义路由逻辑阶梯约束
        if task_type == "creative" or task_type == "logical":
            target_model = "claude" # lmarena.ai
            adapter = self.adapters.get('claude')
        elif task_type == "code" or task_type == "analysis":
            target_model = "gemini" # gemini.google.com
            adapter = self.adapters.get('gemini')
        elif task_type == "search" or task_type == "trending":
            target_model = "grok" # grok.com
            adapter = self.adapters.get('grok')
        else:
            logger.warning(f"Unknown task type: {task_type}. Retrying with Gemini Web...")
            # [逻辑阶梯约束：韧性] 如果某个模型返回错误，系统自动回退尝试另一个模型
            target_model = "gemini"
            adapter = self.adapters.get('gemini')

        if not adapter:
            raise Exception(f"Adapter for {target_model} not registered in The Matrix Router.")

        logger.info(f"Routing Prompt to: {target_model}")
        
        # 准备 JSON 通信握手数据
        handshake_data = self.json_handshake(
            step_id,
            "TheMatrix",
            f"Routing prompt based on task type: {task_type}",
            f"ExecuteAdapter({target_model})",
            {"prompt": prompt}
        )
        
        # 将 Prompt 塞入 Web AI 的输入框 (< 4000 tokens)
        response_data = await adapter.query(prompt)
        
        # 生成最终 JSON 结果握手数据
        return self.json_handshake(
            step_id,
            f"Adapter:{target_model}",
            f"Response received from {target_model}",
            "ResponseComplete",
            {"query": prompt, "response": response_data}
        )
